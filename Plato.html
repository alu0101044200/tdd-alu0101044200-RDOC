def add (lista_alimentos)
    i = 0
    while i < lista_alimentos.size do
        @cantidad.push(100)
        i = i + 1
    end
    @list = lista_alimentos
    @list
end
    add_alimento(alimento, cantidad) click to toggle source

    Método para añadir un alimento a la lista con una cierta cantidad de gramos.

    • Se debe multiplicar cada propiedad del alimento por la cantidad de gramos que se insertan

    • Se sabe que el alimento insertado tiene un peso de 100 gramos

    # File lib/alimentos/plato.rb, line 44
def add_alimento (alimento, cantidad)
    @cantidad.push(cantidad)

    quantity = cantidad/100.0
  
    proteina = alimento.proteina *  quantity
  
    carbohidrato = alimento.carbohidrato * quantity

    lipido = alimento.lipido * quantity

    gei = alimento.gei * quantity

    terreno = alimento.terreno * quantity

    alim = Comida.new(alimento.name.to_s, proteina,carbohidrato,lipido, gei, terreno)

    @list.addmore(alim)
    @list
    
end

def each
    aux = @list.get_head
    if aux == nil
        return nil
    else
        while aux != nil do
            yield aux.value
            aux = aux.next
        end
    end
end

    Devuelve la huella nutricional del plato

    • Segun su valor calorico

    • Segun los gases de efecto invernadero

    Utilizamos Programación funcional

    # File lib/alimentos/plato.rb, line 142
def huella_nutricional()
    energia = valor_calorico_total
    gas = get_gas

    if energia < 670
        t1 =  1
    elsif energia > 830
        t1 = 3 
    else
        t1 = 2
    end

    if gas < 800
        t2 = t1 + 1
    elsif gas > 1200
        t2 = t1 + 3
    else
        t2 = t1 + 2
    end

    (t2.to_f / 2).round(2)

end

    # File lib/alimentos/plato.rb, line 104
def valor_calorico_total
    a = @list.get_proteina
    b = @list.get_carbohidrato
    c = @list.get_lipido
    return (a*4.0 + b*4.0 + c*9.0).round(2)
end

    Se devuelve un string con el nombre de cada alimetno del plato con su respectiva cantidad

    # File lib/alimentos/plato.rb, line 112
def write
    i = 0
    ingedientes = ""
    aux = list.get_head
    y = 0
    while i < list.size do
        ingedientes << aux.value.name + " " + @cantidad[y].to_s + " gramos"
        if aux != list.get_tail
            ingedientes << ", "
        end
        aux = aux.next
        y += 1
        i += 1
    end
 ingedientes
end
