    add(comida, pos) click to toggle source

    Añade un alimento en una determinada posicion de la lista

    • Si la posicion es menos que 0 lo incerta en el head

    • Si la posicion es mayor que el size de la lista lo incerta en el tail

    • Si la posición se encuentra dentro del size d ela lista lo insertará en dicha posicion y movera el resto de los elementos

    # File lib/alimentos/lista.rb, line 70
def add(comida, pos)
    nodo = Node.new(comida,nil,nil)
    if pos >= @size -1
        add_to_tail(comida)
    elsif pos <= 0
        add_to_head(comida)
    else
        @size +=1
        node_r = @head
        i = 0
        while i != pos do
            i +=1
            node_r = node_r.next
        end
        nodo.next = node_r

        aux = node_r.prev
        node_r.prev = nodo

        nodo.prev = aux
        aux.next = nodo
    end
end

def add_to_head(comida)
    nodo = Node.new(comida,nil,nil)
    ini = @head
    if ini == nil
        @size +=1
        @head = nodo
        @tail = nodo

    
    elsif ini =! nil
        @size +=1
        nodo.next = @head
        @head.prev = nodo
        nodo.prev = nil
        @head = nodo
        
    end
end

def addmore(*comida)
    i = 0
    tmp = comida.length
    
    j = 0
    while i < tmp
        i = i +1
        nodo = Node.new(comida.at(j),nil,nil)
        j = j+1
        ini = @head
        if ini == nil
            @size +=1
            @head = nodo
            @tail = nodo
        
        elsif ini =! nil
            @size +=1
            nodo.prev = @tail
            @tail = nodo
            aux = nodo.prev
            aux.next = nodo
        
        end
    end
end

def each
    aux = @head
    if aux == nil
        return nil
    else
        while aux != nil do
            yield aux.value
            aux = aux.next
        end
    end
end

    # File lib/alimentos/lista.rb, line 261
def get_carbohidrato
    aux = @head
    if aux == nil
        return 0
    else
        i = 0
        sum = 0
        while i != @size do
            i += 1
            sum = sum + aux.value.carbohidrato
            aux = aux.next
        end
        sum.round(2)
    end
end
