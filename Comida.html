def initialize(n, p, c, l, g, t)
  # Nombre del alimento
  @name = n
  # cantida de proteeina del aliemnto
  @proteina = p
  # CAntidad de carbono del alimento
  @carbohidrato = c
  # Cantidad de lipidos del alimento
  @lipido = l
  # Cantidad de gas que genera la produccion del alimento
  @gei = g
  # Cantidad de terreno que destroza durante la produccion del alimento

end
