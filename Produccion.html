def get_gases 
    aux = list.get_head
    @gas = 0
   
    while aux != nil do

        @gas = @gas + aux.value.gei
        aux = aux.next

    end
    @gas.round(2)
    
end

def get_terreno
    aux = list.get_head
    @terreno = 0
    while aux != nil do

        @terreno = @terreno + aux.value.terreno
        aux = aux.next

    end
    
    @terreno.round(2)
    
end
