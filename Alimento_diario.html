def requerimiento
   
    if @genero == 0 
        
        if (@datos_finales[0] < 3000) && (@datos_finales[1] < 54)
            return "Calorias insuficientes"
        else
            return "Calorias suficientes"
        end

    end

    if @genero == 1 
        
        if (@datos_finales[0] < 2300) && (@datos_finales[1] < 41)
            return "Calorias insuficientes"
        else
            return "Calorias suficientes"
        end

    end
    
end
